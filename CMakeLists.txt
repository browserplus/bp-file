CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

PROJECT(bpfile)

SET (BPFILE_MAJOR 0)
SET (BPFILE_MINOR 0)
SET (BPFILE_MICRO 1)

SET (BPFILE_DIST_NAME
     "bpfile-${BPFILE_MAJOR}.${BPFILE_MINOR}.${BPFILE_MICRO}")

IF (NOT CMAKE_BUILD_TYPE)
  SET(CMAKE_BUILD_TYPE "Release")
ENDIF (NOT CMAKE_BUILD_TYPE)

IF (${CMAKE_BUILD_TYPE} STREQUAL "Release")
  MESSAGE("** for a debug build: cmake -DCMAKE_BUILD_TYPE=Debug ..")
ENDIF (${CMAKE_BUILD_TYPE} STREQUAL "Release")

SET(CMAKE_C_FLAGS "-Wall")
IF (WIN32) 
  SET(linkFlags "/PDB:NONE /INCREMENTAL:NO /OPT:NOREF /OPT:NOICF")
  SET(CMAKE_EXE_LINKER_FLAGS    "${linkFlags}"
      CACHE STRING "bpfile linker flags" FORCE)
  SET(CMAKE_EXE_LINKER_FLAGS_DEBUG ""
      CACHE STRING "bpfile debug linker flags" FORCE)
  SET(CMAKE_EXE_LINKER_FLAGS_RELEASE
      CACHE STRING "bpfile release linker flags" FORCE)
  SET(CMAKE_SHARED_LINKER_FLAGS "${linkFlags}"
      CACHE STRING "bpfile shared linker flags" FORCE)
  SET(CMAKE_MODULE_LINKER_FLAGS "${linkFlags}"
      CACHE STRING "bpfile module linker flags" FORCE)

  SET(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} /wd4668 /wd4820 /wd4996 /wd4255 /wd4130 /wd4100 /wd4711")
  SET(CMAKE_C_FLAGS_DEBUG "/D DEBUG /Od /Z7")
  SET(CMAKE_C_FLAGS_RELEASE "/D NDEBUG /O2")
  ADD_DEFINITIONS(-DWIN32)
ELSE (WIN32)
  SET(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -fPIC -ansi -isysroot /Developer/SDKs/MacOSX10.4u.sdk -mmacosx-version-min=10.4  -arch i386")
  SET(CMAKE_C_FLAGS_DEBUG "-DDEBUG -g")
  SET(CMAKE_C_FLAGS_RELEASE "-DNDEBUG -O2")
  SET(linkFlags "-mmacosx-version-min=10.4 -isysroot /Developer/SDKs/MacOSX10.4u.sdk -syslibroot,/Developer/SDKs/MacOSX10.4.u.sdk -arch i386")
  SET(CMAKE_EXE_LINKER_FLAGS    "${linkFlags}"
      CACHE STRING "bpfile linker flags" FORCE)
  SET(CMAKE_SHARED_LINKER_FLAGS    "${linkFlags}"
      CACHE STRING "bpfile linker flags" FORCE)
  SET(CMAKE_MODULE_LINKER_FLAGS    "${linkFlags}"
      CACHE STRING "bpfile linker flags" FORCE)
  ADD_DEFINITIONS(-DMACOSX)
ENDIF (WIN32) 

ADD_SUBDIRECTORY(src)
